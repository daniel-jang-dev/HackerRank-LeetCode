### [2566. Maximum Difference by Remapping a Digit](https://leetcode.com/problems/maximum-difference-by-remapping-a-digit/)
#### Brute Force
```Typescript
function minMaxDifference(num: number): number {

  const arr: string[] = String(num).split('');

  // 1. maximum values
  let digitToRemap = '9';
  for (const digit of arr) {
    if (digit !== '9') {
      digitToRemap = digit;
      break;
    }
  }
  const maxValue: number = Number(String(num).replaceAll(digitToRemap, '9'));

  // 2. minimum values
  const firstDigit: string = arr[0];
  const minValue: number = Number(String(num).replaceAll(firstDigit, '0'));

  return maxValue - minValue;
};
```
#### Clean Code
```Typescript
function minMaxDifference(num: number): number {
  const digits = String(num).split('');

  // Max value: replace first non-'9' digit with '9'
  const digitToMax = digits.find(d => d !== '9') ?? '9';
  const maxValue = Number(String(num).replaceAll(digitToMax, '9'));

  // Min value: replace first digit with '0'
  const digitToMin = digits[0];
  const minValue = Number(String(num).replaceAll(digitToMin, '0'));

  return maxValue - minValue;
}
```

```Python3
def min_max_difference(num: int) -> int:
    digits = list(str(num))

    # Max value: replace first non-'9' digit with '9'
    digit_to_max = next((d for d in digits if d != '9'), '9')
    max_value = int(str(num).replace(digit_to_max, '9'))

    # Min value: replace first digit with '0'
    digit_to_min = digits[0]
    min_value = int(str(num).replace(digit_to_min, '0'))

    return max_value - min_value
```
