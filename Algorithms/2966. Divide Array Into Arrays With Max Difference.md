### [2966. Divide Array Into Arrays With Max Difference](https://leetcode.com/problems/divide-array-into-arrays-with-max-difference/)
#### Brute Force
```Typescript
function divideArray(nums: number[], k: number): number[][] {
  nums.sort((a, b) => a - b);
  const result = [];
  let row: number[];
  for (let i = 0; i < nums.length; i++) {
    if (i % 3 === 0) row = [];
    row.push(nums[i]);
    if (i % 3 === 2) {
      if (row[2] - row[0] > k) return [];
      result.push(row);
    }
  }
  return result;
};
```
#### Clean Code
```Typescript
function divideArray(nums: number[], k: number): number[][] {
  nums.sort((a, b) => a - b);
  const result: number[][] = [];

  for (let i = 0; i < nums.length; i += 3) {
    const group = nums.slice(i, i + 3);
    if (group[2] - group[0] > k) return [];
    result.push(group);
  }

  return result;
}
```

```Python3
from typing import List

def divideArray(nums: List[int], k: int) -> List[List[int]]:
    nums.sort()
    result = []

    for i in range(0, len(nums), 3):
        group = nums[i:i + 3]
        if len(group) < 3 or group - group > k:
            return []
        result.append(group)
    return result
```
