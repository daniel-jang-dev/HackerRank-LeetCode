### [3633. Earliest Finish Time for Land and Water Rides I](https://leetcode.com/problems/earliest-finish-time-for-land-and-water-rides-i/)
```Typescript
function earliestFinishTime(
  landStartTime: number[],
  landDuration: number[],
  waterStartTime: number[],
  waterDuration: number[]
): number {
  let minFinishTime = Infinity;

  for (let i = 0; i < landStartTime.length; i++) {
    const landStart = landStartTime[i];
    const landDur = landDuration[i];

    for (let j = 0; j < waterStartTime.length; j++) {
      const waterStart = waterStartTime[j];
      const waterDur = waterDuration[j];

      let finishTime: number;

      if (landStart < waterStart) {
        finishTime = Math.max(landStart + landDur, waterStart) + waterDur;
      } else if (landStart > waterStart) {
        finishTime = Math.max(waterStart + waterDur, landStart) + landDur;
      } else {
        finishTime = landStart + landDur + waterDur;
      }

      minFinishTime = Math.min(minFinishTime, finishTime);
    }
  }

  return minFinishTime;
}
```

```Python3
    land_duration: list[int],
    water_start_time: list[int],
    water_duration: list[int]

    min_finish_time = float('inf')

    for i in range(len(land_start_time)):

        for j in range(len(water_start_time)):
            water_start = water_start_time[j]
            water_dur = water_duration[j]

            if land_start < water_start:
                finish_time = max(land_start + land_dur, water_start) + water_dur
            elif land_start > water_start:
                finish_time = max(water_start + water_dur, land_start) + land_dur
            else:
                finish_time = land_start + land_dur + water_dur

            min_finish_time = min(min_finish_time, finish_time)

    return min_finish_time
```
