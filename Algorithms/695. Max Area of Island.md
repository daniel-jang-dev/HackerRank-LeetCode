### [695. Max Area of Island](https://leetcode.com/problems/max-area-of-island/)
```Typescript
function maxAreaOfIsland(grid: number[][]): number {
    const m = grid.length;
    const n = grid[0].length;
    const visited = grid.map(row => row.map(_ => false));

    let maxArea = 0;
    for (let i = 0; i < m; i++) {
        for (let j = 0; j < n; j++) {
            maxArea = Math.max(maxArea, dfs(i, j));
        }
    }
    return maxArea;

    function dfs(i: number, j: number): number {
        if (i < 0 || i >= m || j < 0 || j >= n) return 0;
        if (visited[i][j]) return 0;
        visited[i][j] = true;
        if (grid[i][j] === 0) return 0;
        return 1 + dfs(i - 1, j) + dfs(i + 1, j) + dfs(i, j - 1) + dfs(i, j + 1);
    }
}
```

```Typescript
/**
 * Calculates the maximum area of an island in the grid.
 * This function mutates the input grid by changing '1's to '0's.
 */
function maxAreaOfIsland(grid: number[][]): number {
  if (!grid || grid.length === 0) {
    return 0;
  }

  const rows = grid.length;
  const cols = grid[0].length;
  let maxArea = 0;

  /**
   * Helper function to find the area of an island using DFS
   * and sink it by changing its '1's to '0's.
   */
  const calculateArea = (r: number, c: number): number => {
    // Base case: check for out-of-bounds or water/visited cells.
    if (r < 0 || r >= rows || c < 0 || c >= cols || grid[r][c] === 0) {
      return 0;
    }

    // Mark the current cell as visited by "sinking" it.
    grid[r][c] = 0;

    // Count the current cell (1) and recursively explore all 4 directions.
    return (
      1 +
      calculateArea(r + 1, c) +
      calculateArea(r - 1, c) +
      calculateArea(r, c + 1) +
      calculateArea(r, c - 1)
    );
  };

  // Iterate through each cell to find islands.
  for (let r = 0; r < rows; r++) {
    for (let c = 0; c < cols; c++) {
      if (grid[r][c] === 1) {
        // If a new island is found, calculate its full area and update maxArea.
        maxArea = Math.max(maxArea, calculateArea(r, c));
      }
    }
  }

  return maxArea;
}
```

```Typescript
function maxAreaOfIsland(grid: number[][]): number {
    const rows = grid.length;
    const cols = grid[0].length;
    const visited = Array.from({ length: rows }, () => Array(cols).fill(false));

    const directions = [[-1, 0], [1, 0], [0, -1], [0, 1]];

    const dfs = (r: number, c: number): number => {
        if (
            r < 0 || r >= rows ||
            c < 0 || c >= cols ||
            visited[r][c] ||
            grid[r][c] === 0
        ) return 0;

        visited[r][c] = true;
        return 1 + directions.reduce((sum, [dr, dc]) => sum + dfs(r + dr, c + dc), 0);
    };

    let maxArea = 0;
    for (let r = 0; r < rows; r++) {
        for (let c = 0; c < cols; c++) {
            maxArea = Math.max(maxArea, dfs(r, c));
        }
    }

    return maxArea;
}
```
