### [2363. Merge Similar Items](https://leetcode.com/problems/merge-similar-items/)
#### Brute Force
```Typescript
function mergeSimilarItems(items1: number[][], items2: number[][]): number[][] {
  const record = new Map();
  for (const [value, weight] of items1) {
    record.set(value,(record.get(value) ?? 0) + weight);
  }
  for (const [value, weight] of items2) {
    record.set(value,(record.get(value) ?? 0) + weight);
  }
  
  const ret = [];
  record.forEach((weight, value) => {
    ret.push([value, weight]);
  });
  ret.sort((a, b) => a[0] - b[0]);
  return ret;
};
```

#### Clean Code
```Typescript
function mergeSimilarItems(items1: number[][], items2: number[][]): number[][] {
    // Use Map to sum weights for each value
    const weights = new Map<number, number>();
    
    // Aggregate weights from items1
    for (const [value, weight] of items1) {
        weights.set(value, (weights.get(value) ?? 0) + weight);
    }
    
    // Aggregate weights from items2
    for (const [value, weight] of items2) {
        weights.set(value, (weights.get(value) ?? 0) + weight);
    }
    
    // Convert to array of [value, weight] pairs and sort by value
    const result: number[][] = Array.from(weights, ([value, weight]) => [value, weight]);
    return result.sort((a, b) => a[0] - b[0]);
}
```

```Python3
def merge_similar_items(items1: list[list[int]], items2: list[list[int]]) -> list[list[int]]:
    from collections import defaultdict

    weight_map = defaultdict(int)

    for value, weight in items1 + items2:
        weight_map[value] += weight

    return sorted([[value, weight] for value, weight in weight_map.items()])
```
