### [3016. Minimum Number of Pushes to Type Word II](https://leetcode.com/problems/minimum-number-of-pushes-to-type-word-ii/)

```Typescript
function minimumPushes(word: string): number {
  let result = 0;
  const charMap = new Map<string, number>();
  for (const c of word) {
    charMap.set(c, (charMap.get(c) ?? 0) + 1);
  }
  console.log(charMap);

  const sortedCounts = [...charMap.values()].sort((a, b) => b - a);
  for (let i = 0; i < sortedCounts.length; i++) {
    const count = sortedCounts[i];
    if (i < 8) {
      result += count;
    } else if (i < 16) {
      result += count * 2;
    } else if (i < 24) {
      result += count * 3;
    } else {
      result += count * 4;
    }
  }
  console.log(sortedCounts);

  return result;
};
```

#### Clean Code
```Typescript
function minimumPushes(word: string): number {
  const frequencyMap = new Map<string, number>();

  for (const char of word) {
    frequencyMap.set(char, (frequencyMap.get(char) ?? 0) + 1);
  }

  const sortedFrequencies = [...frequencyMap.values()].sort((a, b) => b - a);

  return sortedFrequencies.reduce((total, count, index) => {
    const multiplier = Math.floor(index / 8) + 1;
    return total + count * multiplier;
  }, 0);
}
```

```Python3
from collections import Counter

def minimum_pushes(word: str) -> int:
    frequencies = sorted(Counter(word).values(), reverse=True)

    return sum(count * (i // 8 + 1) for i, count in enumerate(frequencies))
```
