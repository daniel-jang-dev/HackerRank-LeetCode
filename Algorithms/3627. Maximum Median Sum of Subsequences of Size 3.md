### [3627. Maximum Median Sum of Subsequences of Size 3](https://leetcode.com/problems/maximum-median-sum-of-subsequences-of-size-3/)
```Typescript
function maximumMedianSum(nums: number[]): number {
  nums.sort((a, b) => a - b);
  const n = nums.length;
  let sum = 0;

  for (let i = n - 2; i >= n / 3; i -= 2) {
    sum += nums[i];
  }

  return sum;
}
```
