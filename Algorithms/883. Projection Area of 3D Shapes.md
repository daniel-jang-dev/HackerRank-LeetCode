### [883. Projection Area of 3D Shapes](https://leetcode.com/problems/projection-area-of-3d-shapes/)
```Typescript
function projectionArea(grid: number[][]): number {
  const n = grid.length;
  let xyArea = 0;
  const xzHighest = new Array<number>(n).fill(0);
  const yzHighest = new Array<number>(n).fill(0);

  for (let i = 0; i < n; i++) {
    for (let j = 0; j < n; j++) {
      const towerHeight = grid[i][j];
      
      // Calculate the area of the XY projection (top view)
      if (towerHeight > 0) {
        xyArea++;
      }

      // Calculate the highest tower in each row (for XZ projection)
      xzHighest[i] = Math.max(xzHighest[i], towerHeight);
      
      // Calculate the highest tower in each column (for YZ projection)
      yzHighest[j] = Math.max(yzHighest[j], towerHeight);
    }
  }

  // Sum up the areas of all three projections
  const xzArea = xzHighest.reduce((sum, height) => sum + height, 0);
  const yzArea = yzHighest.reduce((sum, height) => sum + height, 0);

  return xyArea + xzArea + yzArea;
}
```

```Python3
def projection_area(grid: list[list[int]]) -> int:
    """
    Calculates the total area of the three projections of a 3D grid.

    Args:
        grid: A list of lists of integers representing the height of towers.

    Returns:
        The total area of the three projections.
    """
    n = len(grid)
    xy_area = 0
    xz_highest = [0] * n
    yz_highest = [0] * n

    for i in range(n):
        for j in range(n):
            tower_height = grid[i][j]

            # Calculate the area of the XY projection (top view)
            if tower_height > 0:
                xy_area += 1

            # Calculate the highest tower in each row (for XZ projection)
            xz_highest[i] = max(xz_highest[i], tower_height)

            # Calculate the highest tower in each column (for YZ projection)
            yz_highest[j] = max(yz_highest[j], tower_height)

    # Sum up the areas of all three projections
    xz_area = sum(xz_highest)
    yz_area = sum(yz_highest)

    return xy_area + xz_area + yz_area
```
