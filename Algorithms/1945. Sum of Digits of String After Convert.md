### [1945. Sum of Digits of String After Convert](https://leetcode.com/problems/sum-of-digits-of-string-after-convert)
#### Brute Force
```Typescript
function getLucky(s: string, k: number): number {
    let strNum = '';
    const aCharCode = 'a'.charCodeAt(0);
    for (let i = 0; i < s.length; i++) {
        strNum += String(s.charCodeAt(i) - aCharCode + 1);
    }
    let n = 0;
    while (n < k) {
        const sum = strNum.split('').reduce((a, c) => a + Number(c), 0);
        strNum = String(sum);
        n++;
    }
    return Number(strNum);
};
```
#### Clean Code
```Typescript
function getLucky(s: string, k: number): number {
    const aCode = 'a'.charCodeAt(0);

    // Convert each character to its alphabet position and concatenate
    let strNum = [...s]
        .map(char => String(char.charCodeAt(0) - aCode + 1))
        .join('');

    // Apply digit-sum transformation k times
    for (let i = 0; i < k; i++) {
        const sum = [...strNum].reduce((acc, digit) => acc + Number(digit), 0);
        strNum = String(sum);
    }

    return Number(strNum);
}
```
```Python3
def getLucky(s: str, k: int) -> int:
    # Convert each character to its alphabet position and concatenate
    str_num = ''.join(str(ord(c) - ord('a') + 1) for c in s)

    # Apply digit-sum transformation k times
    for _ in range(k):
        str_num = str(sum(int(d) for d in str_num))

    return int(str_num)
```
