### [2578. Split With Minimum Sum](https://leetcode.com/problems/split-with-minimum-sum/)
#### Brute Force
```Typescript
function splitNum(num: number): number {
  const digits = String(num).split('').sort();
  let num1 = '', num2 = '';
  digits.forEach((d, i) => {
    if (i % 2) {
      num1 += d;
    } else {
      num2 += d;
    }
  });
  return Number(num1) + Number(num2);
};
```
#### Clean Code
```Typescript
function splitNum(num: number): number {
  const digits = [...String(num)].sort();
  const [num1, num2] = digits.reduce<[string, string]>(
    ([a, b], digit, i) => i % 2 ? [a + digit, b] : [a, b + digit],
    ['', '']
  );
  return Number(num1) + Number(num2);
}
```

```Python3
def split_num(num: int) -> int:
    digits = sorted(str(num))
    num1, num2 = zip(*[(d, '') if i % 2 else ('', d) for i, d in enumerate(digits)])
    return int(''.join(num1)) + int(''.join(num2))
```
