#### Clean Code
```Typescript
function getLeastFrequentDigit(n: number): number {
    // Count digit frequencies using a Map
    const frequency = new Map<string, number>();
    for (const digit of String(n)) {
        frequency.set(digit, (frequency.get(digit) || 0) + 1);
    }
    
    // Initialize variables for smallest digit and its count
    let smallestDigit = 9;
    let smallestCount = Number.MAX_SAFE_INTEGER;
    
    // Find the digit with minimum frequency (smallest digit in case of ties)
    for (const [digit, count] of frequency) {
        const numDigit = Number(digit);
        if (count < smallestCount || (count === smallestCount && numDigit < smallestDigit)) {
            smallestDigit = numDigit;
            smallestCount = count;
        }
    }
    
    return smallestDigit;
}
```

```Python3
from typing import Union
from collections import Counter

def get_least_frequent_digit(n: Union[int, float]) -> int:
    # Convert number to string and count digit frequencies
    digit_counts = Counter(str(int(n)))
    
    # Initialize variables for smallest digit and its count
    smallest_digit = 9
    smallest_count = float('inf')
    
    # Find the digit with minimum frequency (smallest digit in case of ties)
    for digit, count in digit_counts.items():
        num_digit = int(digit)
        if count < smallest_count or (count == smallest_count and num_digit < smallest_digit):
            smallest_digit = num_digit
            smallest_count = count
    
    return smallest_digit
```
