### [2309. Greatest English Letter in Upper and Lower Case](https://leetcode.com/problems/greatest-english-letter-in-upper-and-lower-case/)
#### Brute Force
```Typescript
function greatestLetter(s: string): string {

  const seen = new Set<string>();
  const greatest = new Set<string>();

  for (const c of s) {
    const upper = c.toString().toUpperCase();
      const lower = c.toString().toLowerCase();
    if (!seen.has(upper) && !seen.has(lower)) {
      seen.add(c);
    } else {
      if (c === upper && seen.has(lower) || c === lower && seen.has(upper)) {
        greatest.add(upper);
      }
    }
  }
  return [...greatest].sort().reverse()[0] || '';
};
```

#### Clean Code
```Typescript
function greatestLetter(s: string): string {
  const seen = new Set<string>();
  let result = '';

  for (const c of s) {
    const upper = c.toUpperCase();
    const lower = c.toLowerCase();

    seen.add(c);

    if (seen.has(upper) && seen.has(lower)) {
      if (upper > result) {
        result = upper;
      }
    }
  }

  return result;
}
```

```TypeScript
function greatestLetter(s: string): string {
    let chars = Array.from(new Set(s.split("")));
    for (let i = 'Z'.charCodeAt(0); i >= 'A'.charCodeAt(0); i--) {
        const char = String.fromCharCode(i);
        if (chars.includes(char) && chars.includes(char.toLowerCase())) {
            return char
        }
    }
    return ''
};
```

```Python3
def greatest_letter(s: str) -> str:
    seen = set()
    result = ''

    for c in s:
        upper = c.upper()
        lower = c.lower()

        seen.add(c)

        if upper in seen and lower in seen:
            result = max(result, upper)

    return result
```
