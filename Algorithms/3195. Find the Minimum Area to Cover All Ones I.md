### [3195. Find the Minimum Area to Cover All Ones I](https://leetcode.com/problems/find-the-minimum-area-to-cover-all-ones-i/)
```Typescript
function minimumArea(grid: number[][]): number {
  const rows = grid.length;
  const cols = grid[0].length;

  let minRow = rows, maxRow = -1;
  let minCol = cols, maxCol = -1;

  for (let i = 0; i < rows; i++) {
    for (let j = 0; j < cols; j++) {
      if (grid[i][j] !== 0) {
        minRow = Math.min(minRow, i);
        maxRow = Math.max(maxRow, i);
        minCol = Math.min(minCol, j);
        maxCol = Math.max(maxCol, j);
      }
    }
  }

  if (maxRow === -1) return 0; // no non-zero cells

  return (maxRow - minRow + 1) * (maxCol - minCol + 1);
}
```

```Python3
from typing import List

def minimum_area(grid: List[List[int]]) -> int:
    rows, cols = len(grid), len(grid[0])
    min_row, max_row = rows, -1
    min_col, max_col = cols, -1

    for i in range(rows):
        for j in range(cols):
            if grid[i][j] != 0:
                min_row = min(min_row, i)
                max_row = max(max_row, i)
                min_col = min(min_col, j)
                max_col = max(max_col, j)

    if max_row == -1:
        return 0

    return (max_row - min_row + 1) * (max_col - min_col + 1)
```
