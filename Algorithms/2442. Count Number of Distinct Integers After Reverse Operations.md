### [2442. Count Number of Distinct Integers After Reverse Operations](https://leetcode.com/problems/count-number-of-distinct-integers-after-reverse-operations/)
#### Brute Force
```Typescript
function countDistinctIntegers(nums: number[]): number {

  function reverse(n: number): number {
    let reversed = 0;
    while (n) {
      const r = n % 10;
      n = (n - r) / 10;
      reversed = reversed * 10 + r;
    }
    return reversed;
  }

  nums.forEach(n => {
    nums.push(reverse(n));
  });

  const seen = new Set<number>(nums);
  return seen.size;
};
```
#### Clean Code
```Typescript
function countDistinctIntegers(nums: number[]): number {
  const distinctNumbers = new Set<number>();

  // Add original numbers to the set
  for (const num of nums) {
    distinctNumbers.add(num);
  }

  // Add reversed numbers to the set
  for (const num of nums) {
    const reversedNum = parseInt(String(num).split('').reverse().join(''));
    distinctNumbers.add(reversedNum);
  }

  return distinctNumbers.size;
}
```

```Typescript
function countDistinctIntegers(nums: number[]): number {
  // Helper to reverse digits of an integer
  function reverse(n: number): number {
    let reversed = 0;
    while (n > 0) {
      reversed = reversed * 10 + (n % 10);
      n = Math.floor(n / 10);
    }
    return reversed;
  }

  // Create a set to store distinct numbers
  const seen = new Set<number>();

  // Add original and reversed numbers to the set
  for (const n of nums) {
    seen.add(n);
    seen.add(reverse(n));
  }

  return seen.size;
}
```

```Typescript
function countDistinctIntegers(nums: number[]): number {
    function reverseNumber(n: number): number {
        let reversed = 0;
        while (n > 0) {
            reversed = reversed * 10 + (n % 10);
            n = Math.floor(n / 10);
        }
        return reversed;
    }

    const distinctNums = new Set<number>([
        ...nums,
        ...nums.map(reverseNumber)
    ]);

    return distinctNums.size;
}
```

```Python3
from typing import List, Set

def count_distinct_integers(nums: List[int]) -> int:
    def reverse_number(n: int) -> int:
        return int(str(n)[::-1])
    
    # Create a set of original and reversed numbers
    distinct_nums: Set[int] = set(nums)
    distinct_nums.update(reverse_number(n) for n in nums)
    
    return len(distinct_nums)
```
