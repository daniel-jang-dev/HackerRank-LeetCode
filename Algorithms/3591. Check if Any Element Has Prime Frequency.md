### [3591. Check if Any Element Has Prime Frequency](https://leetcode.com/problems/check-if-any-element-has-prime-frequency/)
```Typescript
function checkPrimeFrequency(nums: number[]): boolean {
  function isPrime(n: number): boolean {
    if (n < 2) return false;
    const limit = Math.floor(Math.sqrt(n));
    for (let i = 2; i <= limit; i++) {
      if (n % i === 0) return false;
    }
    return true;
  }

  const counter = new Map<number, number>();
  nums.forEach(n => {
    counter.set(n, (counter.get(n) || 0) + 1);
  });

  for (const count of counter.values()) {
    if (isPrime(count)) return true;
  }

  return false;
}
```

```Python3
from collections import Counter
import math

def is_prime(n: int) -> bool:
    if n < 2:
        return False
    limit = int(math.sqrt(n))
    for i in range(2, limit + 1):
        if n % i == 0:
            return False
    return True

def check_prime_frequency(nums: list[int]) -> bool:
    freq = Counter(nums)
    for count in freq.values():
        if is_prime(count):
            return True
    return False
```
