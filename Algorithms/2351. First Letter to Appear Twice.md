### [2351. First Letter to Appear Twice](https://leetcode.com/problems/first-letter-to-appear-twice/)
```Typescript
function repeatedCharacter(s: string): string {
  const seen = new Set<string>();
  for (const char of s) {
    if (seen.has(char)) {
      return char;
    }
    seen.add(char);
  }
  return '';
}
```
```Python3
def repeated_character(s: str) -> str:
    seen = set()
    for char in s:
        if char in seen:
            return char
        seen.add(char)
    return ''
```
